## Installation

```bash
npm install @your-org/ui
```

## Usage

```tsx
import { Accordion } from "@/component/ui/accordion";

function App() {
  return (
    <Accordion>
      <Accordion.Item value="item-1" label="First Item">
        This is the content of the first item
      </Accordion.Item>
      <Accordion.Item value="item-2" label="Second Item">
        This is the content of the second item
      </Accordion.Item>
    </Accordion>
  );
}
```

## Examples

### Basic Usage

Only one item can be expanded at a time. The first item is not expanded by default.

```tsx
<Accordion>
  <Accordion.Item value="item-1" label="What is React?">
    React is a JavaScript library for building user interfaces.
  </Accordion.Item>
  <Accordion.Item value="item-2" label="What is TypeScript?">
    TypeScript is a superset of JavaScript that adds a type system.
  </Accordion.Item>
  <Accordion.Item value="item-3" label="What is a Component?">
    A component is a reusable UI snippet that can accept inputs and return React
    elements.
  </Accordion.Item>
</Accordion>
```

### Multiple Expansion

Use the `allowMultiple` prop to allow multiple items to be expanded simultaneously.

```tsx
<Accordion allowMultiple>
  <Accordion.Item value="item-1" label="What is React?">
    React is a JavaScript library for building user interfaces.
  </Accordion.Item>
  <Accordion.Item value="item-2" label="What is TypeScript?">
    TypeScript is a superset of JavaScript that adds a type system.
  </Accordion.Item>
</Accordion>
```

### Controlled Mode

Implement controlled mode through `openKeys` and `onOpenChange`.

```tsx
const [openKeys, setOpenKeys] = useState(["item-1"]);

<Accordion openKeys={openKeys} onOpenChange={setOpenKeys}>
  <Accordion.Item value="item-1" label="First Item">
    Content
  </Accordion.Item>
  <Accordion.Item value="item-2" label="Second Item">
    Content
  </Accordion.Item>
</Accordion>;
```

### Custom Styling

Customize container styling through `className`, and customize internal wrapped element styling through `itemProps`.

```tsx
<Accordion className="max-w-md">
  <Accordion.Item
    value="item-1"
    label="Custom Styling"
    itemProps={{
      root: { className: "bg-blue-50" },
      content: { className: "bg-blue-50/50" },
    }}
  >
    Content Area
  </Accordion.Item>
</Accordion>
```

### Custom Icon

Customize the expand/collapse icon through the `icon` prop. It supports both static nodes and function forms.

```tsx
import { Plus } from "lucide-react";

<Accordion icon={<Plus className="w-4 h-4" />}>
  <Accordion.Item value="item-1" label="What is React?">
    <p>React is a JavaScript library for building user interfaces.</p>
  </Accordion.Item>
  <Accordion.Item value="item-2" label="Why Use React?">
    <p>React uses component-based development, data-driven views, and has a rich ecosystem.</p>
  </Accordion.Item>
</Accordion>
```

**Using function form (different icons for open/close state):**

```tsx
<Accordion icon={(open) => open ? <Minus /> : <Plus />}>
  <Accordion.Item value="item-1" label="First Item">
    Content
  </Accordion.Item>
</Accordion>
```

## API Reference

### Accordion Props

| Prop            | Type                     | Default | Description                                                  |
| --------------- | ------------------------ | ------- | ------------------------------------------------------------ |
| defaultOpenKeys | string[]                 | []      | Initially expanded keys (uncontrolled)                       |
| openKeys        | string[]                 | -       | Currently expanded keys (controlled)                         |
| onOpenChange    | (keys: string[]) => void | -       | Callback when expansion state changes                        |
| allowMultiple   | boolean                  | false   | Whether to allow multiple expansions                         |
| className       | ClassNameValue           | -       | `string\|string[]\|boolean\|null\|undefined\|Record<string,string\|string[]>` |
| icon            | ReactNode \| ((open: boolean) => ReactNode) | -       | Custom expand/collapse icon                                  |

### Accordion.Item Props

| Prop      | Type           | Default | Description                                                  |
| --------- | -------------- | ------- | ------------------------------------------------------------ |
| value     | string         | -       | Unique identifier (required)                                 |
| label     | ReactNode      | -       | Label content (required)                                     |
| className | ClassNameValue | -       | `string\|string[]\|boolean\|null\|undefined\|Record<string,string\|string[]>` |
| itemProps | object         | -       | Internal wrapped props pass-through config                   |
| icon      | ReactNode \| ((open: boolean) => ReactNode) | -       | Custom icon for this item (overrides Accordion icon)        |

### itemProps Config

| Property | Type                           | Description          |
| -------- | ------------------------------ | -------------------- |
| root     | React.ComponentProps<"div">    | Root element props   |
| label    | React.ComponentProps<"span">   | Label element props  |
| content  | React.ComponentProps<"div">    | Content area props  |
