## Installation

```bash
npm install @your-org/ui
```

## Usage

```tsx
import { Checkbox } from "@/component";
```

## Examples

### Basic

```tsx
<Checkbox
  label="Basic Checkbox"
  options={[
    { value: "a", label: "Option A" },
    { value: "b", label: "Option B" },
  ]}
/>
```

### Controlled

```tsx
const [value, setValue] = useState(["a"]);
<Checkbox
  label="Controlled Checkbox"
  value={value}
  onValueChange={setValue}
  options={[
    { value: "a", label: "Option A" },
    { value: "b", label: "Option B" },
  ]}
/>
```

### Disabled

```tsx
<Checkbox
  label="Disabled Checkbox"
  disabled
  options={[
    { value: "a", label: "Option A" },
    { value: "b", label: "Option B" },
  ]}
/>
```

### Variants

```tsx
<Checkbox
  label="Toggle Variant"
  options={[
    { value: "a", label: "Toggle A", indicator: { hidden: true } },
    { value: "b", label: "Toggle B", indicator: { hidden: true } },
  ]}
  slotProps={{
    options: {
      variant: "toggle",
    },
  }}
/>
```

## API Reference

### Checkbox Props

| Prop           | Type                                        | Default   | Description                                                                              |
| ---------------|---------------------------------------------|-----------|------------------------------------------------------------------------------------------|
| defaultValue   | T[]                                         | []        | Initial selected values (uncontrolled)                                                   |
| value          | T[]                                         | -         | Current selected values (controlled)                                                     |
| onValueChange  | (value: T[]) => void \| { invalid?: string } | -         | Value change callback, can return { invalid?: string } to trigger error state          |
| invalid        | boolean \| string                           | -         | Error state, supports boolean or string type                                            |
| label          | ReactNode                                   | -         | Label text                                                                              |
| description    | ReactNode                                   | -         | Description text                                                                        |
| disabled       | boolean                                     | false     | Disable entire checkbox group                                                          |
| name           | string                                      | -         | Form input name                                                                         |
| options        | CheckboxItemProps[]                         | -         | Checkbox options array                                                                 |

### CheckboxItemProps

| Prop            | Type                       | Default    | Description                                                     |
| ----------------|----------------------------|------------|-----------------------------------------------------------------|
| value           | string                     | -          | Unique identifier                                              |
| label           | ReactNode                  | -          | Display text                                                   |
| disabled        | boolean                    | false      | Disable individual checkbox                                    |
| variant         | "toggle" \| "checkbox"     | "checkbox" | Style variant                                                  |
| indicator       | object                     | -          | Indicator config                                               |

### indicator Config

| Property  | Type                                                                         | Default         | Description                                                                                    |
| --------- | ---------------------------------------------------------------------------- | --------------- | ---------------------------------------------------------------------------------------------- |
| checked   | ReactNode                                                                    | \<CheckIcon /\> | Indicator when checked                                                                         |
| unchecked | ReactNode                                                                    | -               | Indicator when unchecked                                                                       |
| hidden    | boolean                                                                      | false           | Hide indicator                                                                                 |
| className | ClassNameValue                                                               | -               | Custom indicator class name                                                                    |
| props     | Omit\<ComponentProps\<"span"\>, "className"\> & { className?: ClassNameValue } | -               | Indicator element props pass-through                                                            |

### slotProps Config

| Property    | Type                              | Description                              |
| ------------|-----------------------------------|------------------------------------------|
| root        | ComponentProps\<"div">            | Root container `<div>` element props     |
| content     | ComponentProps\<"div">            | Content container `<div>` element props |
| label       | ComponentProps\<"label">          | Label `<label>` element props            |
| description | ComponentProps\<"small">          | Description `<small>` element props      |
| invalid     | ComponentProps\<"span">           | Error text `<span>` element props        |
| options     | Omit\<CheckboxItemProps, "checked" \| "value" \| "label"> | Checkbox item element props |
