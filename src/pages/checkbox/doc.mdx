## Installation

```bash
npm install @your-org/ui
```

## Usage

```tsx
import { Checkbox } from "@/component/ui/checkbox";

function App() {
  return (
    <Checkbox.Group>
      <Checkbox value="1">Option 1</Checkbox>
      <Checkbox value="2">Option 2</Checkbox>
    </Checkbox.Group>
  );
}
```

## Examples

### Basic

```tsx
<Checkbox value="agree">I agree</Checkbox>
```

### Controlled

```tsx
const [checked, setChecked] = useState(["1"]);
<Checkbox.Group value={checked} onValueChange={setChecked}>
  <Checkbox value="1">Option 1</Checkbox>
  <Checkbox value="2">Option 2</Checkbox>
</Checkbox.Group>
```

### Direction

```tsx
<Checkbox.Group direction="horizontal">
  <Checkbox value="1">Option 1</Checkbox>
  <Checkbox value="2">Option 2</Checkbox>
</Checkbox.Group>

<Checkbox.Group direction="vertical">
  <Checkbox value="1">Option 1</Checkbox>
  <Checkbox value="2">Option 2</Checkbox>
</Checkbox.Group>
```

### Disabled

```tsx
<Checkbox disabled>Disabled</Checkbox>
<Checkbox.Group disabled>
  <Checkbox value="1">Disabled Group</Checkbox>
</Checkbox.Group>
```

### Variants

```tsx
<Checkbox variant="text">Text</Checkbox>
<Checkbox variant="primary">Primary</Checkbox>
<Checkbox variant="secondary">Secondary</Checkbox>
<Checkbox variant="outlined">Outlined</Checkbox>
```

## API Reference

### Checkbox.Group Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| defaultValue | T[] | - | Initial selected values (uncontrolled) |
| value | T[] | - | Current selected values (controlled) |
| onValueChange | (value: T[]) => void | - | Value change callback |
| direction | "horizontal" \| "vertical" | "horizontal" | Layout direction |
| disabled | boolean | false | Disable entire group |
| className | ClassNameValue | - | Custom class name |
| children | ReactNode | - | Child elements |

### Checkbox Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| value | string | - | Unique identifier (required) |
| variant | "text" \| "primary" \| "secondary" \| "outlined" | "text" | Style variant |
| disabled | boolean | false | Disable individual checkbox |
| indicator | object | - | Indicator config |
| children | ReactNode | - | Display text |
| className | ClassNameValue | - | Custom class name |

### indicator Config

| Property | Type | Default | Description |
|------|------|---------|------|
| checked | ReactNode | \<CheckIcon /\> | Indicator when checked |
| unchecked | ReactNode | \<SquareIcon /\> | Indicator when unchecked |
| hidden | boolean | false | Hide indicator |
| className | ClassNameValue | - | Custom indicator class name |
| props | React.ComponentProps<"span"> | - | Indicator element props pass-through |
