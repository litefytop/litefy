```bash
npm install @your-org/ui
```

## 用法

```tsx
import { Checkbox } from "@/component/ui/checkbox";

function App() {
  return (
    <Checkbox.Group>
      <Checkbox value="1">Option 1</Checkbox>
      <Checkbox value="2">Option 2</Checkbox>
    </Checkbox.Group>
  );
}
```

## 示例

### 基础用法

```tsx
<Checkbox value="agree">I agree</Checkbox>
```

### 受控模式

```tsx
const [checked, setChecked] = useState(["1"]);
<Checkbox.Group value={checked} onValueChange={setChecked}>
  <Checkbox value="1">Option 1</Checkbox>
  <Checkbox value="2">Option 2</Checkbox>
</Checkbox.Group>
```

### 方向

```tsx
<Checkbox.Group direction="horizontal">
  <Checkbox value="1">Option 1</Checkbox>
  <Checkbox value="2">Option 2</Checkbox>
</Checkbox.Group>

<Checkbox.Group direction="vertical">
  <Checkbox value="1">Option 1</Checkbox>
  <Checkbox value="2">Option 2</Checkbox>
</Checkbox.Group>
```

### 禁用状态

```tsx
<Checkbox disabled>Disabled</Checkbox>
<Checkbox.Group disabled>
  <Checkbox value="1">Disabled Group</Checkbox>
</Checkbox.Group>
```

### 样式变体

```tsx

<Checkbox variant="primary">Primary</Checkbox>
<Checkbox variant="secondary">Secondary</Checkbox>
```

## API Reference

### Checkbox.Group Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| defaultValue | T[] | - | 初始选中的值（非受控模式） |
| value | T[] | - | 当前选中的值（受控模式） |
| onValueChange | (value: T[]) => void | - | 值变化回调 |
| direction | "horizontal" \| "vertical" | "horizontal" | 排列方向 |
| disabled | boolean | false | 是否禁用整组 |
| className | ClassNameValue | - | 自定义类名 |
| children | ReactNode | - | 子元素 |

### Checkbox Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| value | string | - | 唯一标识符（必填） |
| variant | "text" \| "primary" \| "secondary" \| "outlined" | "text" | 样式变体 |
| disabled | boolean | false | 是否禁用单个 |
| indicator | object | - | 指示器配置 |
| children | ReactNode | - | 显示文本 |
| className | ClassNameValue | - | 自定义类名 |

### indicator 配置

| 属性 | 类型 | 默认值 | 描述 |
|------|------|---------|------|
| checked | ReactNode | \<CheckIcon /\> | 选中时的指示器 |
| unchecked | ReactNode | \<SquareIcon /\> | 未选中时的指示器 |
| hidden | boolean | false | 是否隐藏指示器 |
| className | ClassNameValue | - | 指示器自定义类名 |
| props | React.ComponentProps<"span"> | - | 指示器元素属性透传 |
