## Installation

```bash
npm install @your-org/ui
```

## Usage

```tsx
import { AccordionControlled } from "@/component/ui/accordion-controlled";

function App() {
  const [openKeys, setOpenKeys] = useState<string[]>([]);
  
  return (
    <AccordionControlled openKeys={openKeys} onOpenChange={setOpenKeys}>
      <AccordionControlled.Item value="item-1" label="First Item">
        This is the content of the first item
      </AccordionControlled.Item>
      <AccordionControlled.Item value="item-2" label="Second Item">
        This is the content of the second item
      </AccordionControlled.Item>
    </AccordionControlled>
  );
}
```

## Examples

### Basic Usage

Basic usage with controlled mode.

```tsx
const [openKeys, setOpenKeys] = useState(["item-1"]);

<AccordionControlled openKeys={openKeys} onOpenChange={setOpenKeys}>
  <AccordionControlled.Item value="item-1" label="What is React?">
    React is a JavaScript library for building user interfaces.
  </AccordionControlled.Item>
  <AccordionControlled.Item value="item-2" label="What is TypeScript?">
    TypeScript is a superset of JavaScript that adds a type system.
  </AccordionControlled.Item>
</AccordionControlled>;
```

### Controlled Mode

Use `openKeys` and `onOpenChange` to implement controlled mode, allowing you to manage the expanded state programmatically.

```tsx
const [openKeys, setOpenKeys] = useState(["item-1"]);

<div className="space-y-4">
  <div>Current open: {openKeys.join(", ")}</div>
  <AccordionControlled
    openKeys={openKeys}
    onOpenChange={setOpenKeys}
  >
    <AccordionControlled.Item value="item-1" label="First Item">
      Content
    </AccordionControlled.Item>
    <AccordionControlled.Item value="item-2" label="Second Item">
      Content
    </AccordionControlled.Item>
  </AccordionControlled>
</div>;
```

### Multiple Expansion

Use the `allowMultiple` prop to allow multiple items to be expanded simultaneously.

```tsx
const [openKeys, setOpenKeys] = useState(["item-1", "item-2"]);

<AccordionControlled
  openKeys={openKeys}
  onOpenChange={setOpenKeys}
  allowMultiple
>
  <AccordionControlled.Item value="item-1" label="First Item">
    Content
  </AccordionControlled.Item>
  <AccordionControlled.Item value="item-2" label="Second Item">
    Content
  </AccordionControlled.Item>
</AccordionControlled>;
```

### Custom Icon

Customize the expand/collapse icon through the `icon` prop. It supports both static nodes and function forms.

```tsx
const [openKeys, setOpenKeys] = useState([]);

<AccordionControlled
  openKeys={openKeys}
  onOpenChange={setOpenKeys}
  icon={(open) => <Plus className={`w-4 h-4 ${open ? "rotate-45" : ""}`} />}
>
  <AccordionControlled.Item value="item-1" label="Custom Icon">
    Content
  </AccordionControlled.Item>
</AccordionControlled>;
```

## API

### AccordionControlled Props

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `defaultOpenKeys` | `string[]` | `[]` | Default expanded keys |
| `openKeys` | `string[]` | - | Controlled expanded keys |
| `onOpenChange` | `(keys: string[]) => void` | - | Callback when expanded keys change |
| `allowMultiple` | `boolean` | `false` | Allow multiple items to be expanded |
| `className` | `ClassNameValue` | - | Container class name |
| `icon` | `ReactNode \| ((open: boolean) => ReactNode)` | - | Custom expand/collapse icon |

### AccordionControlled.Item Props

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `value` | `string` | - | Unique identifier (required) |
| `label` | `ReactNode` | - | Item title content (required) |
| `className` | `ClassNameValue` | - | Item class name |
| `itemProps` | `object` | - | Internal element props passthrough |
| `icon` | `ReactNode \| ((open: boolean) => ReactNode)` | - | Custom icon (overrides Accordion's icon) |

### itemProps Configuration

| Prop | Type | Description |
| --- | --- | --- |
| `root` | `React.ComponentProps<"div">` | Root element props |
| `label` | `React.ComponentProps<"span">` | Label element props |
| `content` | `React.ComponentProps<"div">` | Content area props |
