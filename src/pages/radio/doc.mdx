## Installation

```bash
npm install @your-org/ui
```

## Usage

```tsx
import { Radio } from "@/component/ui/radio";

function App() {
  return (
    <Radio.Group>
      <Radio value="1">Option 1</Radio>
      <Radio value="2">Option 2</Radio>
    </Radio.Group>
  );
}
```

## Examples

### Basic

```tsx
<Radio.Group>
  <Radio value="1">Option 1</Radio>
  <Radio value="2">Option 2</Radio>
  <Radio value="3">Option 3</Radio>
</Radio.Group>
```

### Controlled

```tsx
const [value, setValue] = useState("1");
<Radio.Group value={value} onValueChange={setValue}>
  <Radio value="1">Option 1</Radio>
  <Radio value="2">Option 2</Radio>
</Radio.Group>
```

### Direction

```tsx
<Radio.Group direction="horizontal">
  <Radio value="1">Option 1</Radio>
  <Radio value="2">Option 2</Radio>
</Radio.Group>

<Radio.Group direction="vertical">
  <Radio value="1">Option 1</Radio>
  <Radio value="2">Option 2</Radio>
</Radio.Group>
```

### Disabled

```tsx
<Radio disabled>Disabled</Radio>
<Radio.Group disabled>
  <Radio value="1">Disabled Group</Radio>
</Radio.Group>
```

### Variants

```tsx
<Radio variant="text">Text</Radio>
<Radio variant="primary">Primary</Radio>
<Radio variant="secondary">Secondary</Radio>
<Radio variant="outlined">Outlined</Radio>
```

## API Reference

### Radio.Group Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| defaultValue | T | - | Initial selected value (uncontrolled) |
| value | T | - | Current selected value (controlled) |
| onValueChange | (value: T) => void | - | Value change callback |
| direction | "horizontal" \| "vertical" | "horizontal" | Layout direction |
| disabled | boolean | false | Disable entire group |
| invalid | boolean | false | Invalid state |
| className | ClassNameValue | - | Custom class name |

### Radio Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| value | string | - | Unique identifier (required) |
| variant | "text" \| "primary" \| "secondary" \| "outlined" | "text" | Style variant |
| disabled | boolean | false | Disable individual radio |
| indicator | object | - | Indicator config |
| children | ReactNode | - | Display text |
| className | ClassNameValue | - | Custom class name |

### indicator Config

| Property | Type | Default | Description |
|------|------|---------|------|
| checked | ReactNode | - | Indicator when checked |
| unchecked | ReactNode | - | Indicator when unchecked |
| hidden | boolean | false | Hide indicator |
| className | ClassNameValue | - | Custom indicator class name |
| props | React.ComponentProps<"span"> | - | Indicator element props pass-through |
